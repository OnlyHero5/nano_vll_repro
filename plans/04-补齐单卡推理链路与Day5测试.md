## 2. 先修知识

## 2.1 为什么要区分 Prefill 和 Decode

这是 vLLM / nano-vLLM 最核心的运行时思想之一。

### Prefill

输入完整 prompt，一次性算出 prompt 里所有 token 的 attention。

特点：

- 输入 shape 大
- 计算量大
- 更适合直接 eager 执行

### Decode

每个 step 每条序列只输入 1 个新 token，但要读完整 KV Cache。

特点：

- 输入 shape 小
- 会重复执行很多次
- 非常适合后续做 CUDA Graph

## 2.2 为什么 `Scheduler` 不应该自己管模型执行

`Scheduler` 的职责是“决定这一轮谁该跑”，不是“怎么跑模型”。

职责边界应该是：

- `Scheduler`：队列、状态、抢占、完成判断
- `ModelRunner`：输入准备、上下文构建、模型前向、采样
- `LLMEngine`：把上面两者串起来

这个边界会直接决定你 Day6 的 TP 和 CUDA Graph 好不好接。

```python
from typing import Optional

import torch
from torch import nn

from config import Config
from engine.sequence import Sequence
from layers.linear import divide, get_tp_rank, get_tp_world_size
from layers.sampler import Sampler
from utils.context import Context, get_context, set_context
from utils.loader import load_model


class ModelRunner:
    def __init__(self, config: Config):
        self.config = config
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.model_dtype = config.torch_dtype
        self.kv_cache_dtype = config.kv_torch_dtype

        self.tp_rank = get_tp_rank()
        self.tp_size = get_tp_world_size()

        Sequence.block_size = config.kvcache_block_size
        self.block_size = Sequence.block_size

        self.model = self._load_model()
        self.sampler = Sampler()
        self.kv_cache: Optional[list[torch.Tensor]] = None

        self.num_layers = self.model.config.num_hidden_layers
        self.head_dim = getattr(
            self.model.config,
            "head_dim",
            self.model.config.hidden_size // self.model.config.num_attention_heads,
        )
        self.num_kv_heads = divide(self.model.config.num_key_value_heads, self.tp_size)

    def _load_model(self) -> nn.Module:
        from models.qwen3 import Qwen3ForCausalLM

        print(f"[ModelRunner] 加载模型：{self.config.model_path}")
        model = Qwen3ForCausalLM.from_pretrained(self.config.model_path)
        load_model(model, self.config.model_path)
        model = model.to(self.device, dtype=self.model_dtype)
        model.eval()
        return model

    def get_num_free_gpu_blocks(self) -> int:
        if self.config.num_kvcache_blocks > 0:
            return self.config.num_kvcache_blocks

        if not torch.cuda.is_available():
            return 128

        total_memory = torch.cuda.get_device_properties(self.device).total_memory
        allocated_memory = torch.cuda.memory_allocated(self.device)
        free_memory = total_memory - allocated_memory
        available_memory = int(free_memory * self.config.gpu_memory_utilization)

        bytes_per_block_per_layer = (
            2
            * self.block_size
            * self.num_kv_heads
            * self.head_dim
            * torch.tensor([], dtype=self.kv_cache_dtype).element_size()
        )
        bytes_per_block = bytes_per_block_per_layer * self.num_layers
        num_blocks = max(1, available_memory // bytes_per_block)

        print(f"[ModelRunner] 可分配 KV Cache 块数：{num_blocks}")
        return int(num_blocks)

    def allocate_kv_cache(self, num_blocks: int) -> None:
        self.kv_cache = []
        for _ in range(self.num_layers):
            cache = torch.zeros(
                2,
                num_blocks,
                self.block_size,
                self.num_kv_heads,
                self.head_dim,
                dtype=self.kv_cache_dtype,
                device=self.device,
            )
            self.kv_cache.append(cache)

        print(f"[ModelRunner] KV Cache 分配完成：{num_blocks} blocks x {self.num_layers} layers")

    def prepare_prefill(self, sequences: list[Sequence]) -> tuple[torch.Tensor, torch.Tensor]:
        all_token_ids = []
        all_positions = []
        cu_seqlens = [0]
        slot_mapping = []

        for seq in sequences:
            seq_len = len(seq.token_ids)
            all_token_ids.extend(seq.token_ids)
            all_positions.extend(range(seq_len))
            cu_seqlens.append(cu_seqlens[-1] + seq_len)

            for i in range(seq_len):
                block_idx = i // self.block_size
                offset = i % self.block_size
                block_id = seq.block_table[block_idx]
                slot_mapping.append(block_id * self.block_size + offset)

        input_ids = torch.tensor(all_token_ids, dtype=torch.long, device=self.device)
        positions = torch.tensor(all_positions, dtype=torch.long, device=self.device)
        max_seqlen = max(len(seq.token_ids) for seq in sequences)

        context = Context(
            is_prefill=True,
            cu_seqlens_q=torch.tensor(cu_seqlens, dtype=torch.int32, device=self.device),
            cu_seqlens_k=torch.tensor(cu_seqlens, dtype=torch.int32, device=self.device),
            max_seqlen_q=max_seqlen,
            max_seqlen_k=max_seqlen,
            slot_mapping=torch.tensor(slot_mapping, dtype=torch.long, device=self.device),
            context_lens=None,
            block_tables=None,
            max_context_len=None,
            max_num_blocks=None,
            kv_cache=self.kv_cache,
        )
        set_context(context)
        return input_ids, positions

    def prepare_decode(self, sequences: list[Sequence]) -> tuple[torch.Tensor, torch.Tensor]:
        input_ids = []
        positions = []
        context_lens = []
        block_tables = []
        slot_mapping = []

        max_num_blocks = max(len(seq.block_table) for seq in sequences)

        for seq in sequences:
            input_ids.append(seq.last_token)
            positions.append(seq.num_tokens - 1)
            context_lens.append(seq.num_tokens)

            padded_block_table = seq.block_table.copy()
            while len(padded_block_table) < max_num_blocks:
                padded_block_table.append(0)
            block_tables.append(padded_block_table)

            pos = seq.num_tokens - 1
            block_idx = pos // self.block_size
            offset = pos % self.block_size
            block_id = seq.block_table[block_idx]
            slot_mapping.append(block_id * self.block_size + offset)

        input_ids = torch.tensor(input_ids, dtype=torch.long, device=self.device)
        positions = torch.tensor(positions, dtype=torch.long, device=self.device)

        context = Context(
            is_prefill=False,
            cu_seqlens_q=None,
            cu_seqlens_k=None,
            max_seqlen_q=None,
            max_seqlen_k=None,
            slot_mapping=torch.tensor(slot_mapping, dtype=torch.long, device=self.device),
            context_lens=torch.tensor(context_lens, dtype=torch.int32, device=self.device),
            block_tables=torch.tensor(block_tables, dtype=torch.int32, device=self.device),
            max_context_len=max(context_lens),
            max_num_blocks=max_num_blocks,
            kv_cache=self.kv_cache,
        )
        set_context(context)
        return input_ids, positions

    @torch.inference_mode()
    def run(self, sequences: list[Sequence], is_prefill: bool) -> list[int]:
        if not sequences:
            return []

        if is_prefill:
            input_ids, positions = self.prepare_prefill(sequences)
        else:
            input_ids, positions = self.prepare_decode(sequences)

        logits = self.model(input_ids, positions)

        if is_prefill:
            context = get_context()
            last_token_indices = (context.cu_seqlens_q[1:] - 1).long()
            logits = logits[last_token_indices]

        temperatures = torch.tensor(
            [seq.temperature for seq in sequences],
            dtype=torch.float32,
            device=self.device,
        )
        top_ks = torch.tensor(
            [getattr(seq, "top_k", 0) for seq in sequences],
            dtype=torch.long,
            device=self.device,
        )
        top_ps = torch.tensor(
            [getattr(seq, "top_p", 1.0) for seq in sequences],
            dtype=torch.float32,
            device=self.device,
        )

        next_tokens = self.sampler(logits, temperatures, top_ks, top_ps)
        return next_tokens.tolist()
```

### 5.2 替换 `engine/llm_engine.py`

```python
import atexit
from time import perf_counter
from typing import Union

import torch
from tqdm.auto import tqdm
from transformers import AutoTokenizer

from config import Config
from engine.block_manager import BlockManager
from engine.model_runner import ModelRunner
from engine.scheduler import Scheduler
from engine.sequence import Sequence
from sampling_params import SamplingParams


class LLMEngine:
    def __init__(self, model: str, **kwargs):
        self.config = Config(model_path=model, **kwargs)
        Sequence.block_size = self.config.kvcache_block_size

        self.tokenizer = AutoTokenizer.from_pretrained(
            self.config.model_path,
            trust_remote_code=True,
        )
        self.config.eos = self.tokenizer.eos_token_id

        self.model_runner = ModelRunner(self.config)

        num_blocks = self.model_runner.get_num_free_gpu_blocks()
        num_blocks = max(1, int(num_blocks * 0.95))
        self.model_runner.allocate_kv_cache(num_blocks)

        self.block_manager = BlockManager(num_blocks, Sequence.block_size)
        self.scheduler = Scheduler(self.config, self.block_manager)

        atexit.register(self._cleanup)

    def _cleanup(self) -> None:
        if torch.cuda.is_available():
            torch.cuda.empty_cache()

    def add_request(
        self,
        prompt: Union[str, list[int]],
        sampling_params: SamplingParams | None = None,
    ) -> None:
        if sampling_params is None:
            sampling_params = SamplingParams()

        if isinstance(prompt, str):
            token_ids = self.tokenizer.encode(prompt)
        else:
            token_ids = list(prompt)

        seq = Sequence(token_ids, sampling_params)
        seq.top_k = sampling_params.top_k
        seq.top_p = sampling_params.top_p
        self.scheduler.add_sequence(seq)

    def is_finished(self) -> bool:
        return self.scheduler.is_finished()

    def step(self) -> tuple[list[tuple[int, list[int]]], int]:
        seqs, is_prefill = self.scheduler.schedule()
        if not seqs:
            return [], 0

        token_ids = self.model_runner.run(seqs, is_prefill)
        finished_seqs = self.scheduler.postprocess(seqs, token_ids)

        outputs = [(seq.seq_id, seq.completion_token_ids) for seq in finished_seqs]
        num_tokens = sum(len(seq) for seq in seqs) if is_prefill else -len(seqs)
        return outputs, num_tokens

    def generate(
        self,
        prompts: Union[list[str], list[list[int]]],
        sampling_params: Union[SamplingParams, list[SamplingParams], None] = None,
        use_tqdm: bool = True,
    ) -> list[dict]:
        if sampling_params is None:
            sampling_params = SamplingParams()

        if not isinstance(sampling_params, list):
            sampling_params = [sampling_params] * len(prompts)

        for prompt, sp in zip(prompts, sampling_params):
            self.add_request(prompt, sp)

        outputs = {}
        pbar = tqdm(total=len(prompts), desc="Generating", dynamic_ncols=True) if use_tqdm else None

        prefill_throughput = 0.0
        decode_throughput = 0.0

        while not self.is_finished():
            start = perf_counter()
            output, num_tokens = self.step()
            elapsed = perf_counter() - start

            if pbar is not None and elapsed > 0:
                if num_tokens > 0:
                    prefill_throughput = num_tokens / elapsed
                elif num_tokens < 0:
                    decode_throughput = -num_tokens / elapsed
                pbar.set_postfix(
                    {
                        "Prefill": f"{int(prefill_throughput)} tok/s",
                        "Decode": f"{int(decode_throughput)} tok/s",
                    }
                )

            for seq_id, token_ids in output:
                outputs[seq_id] = token_ids
                if pbar is not None:
                    pbar.update(1)

        if pbar is not None:
            pbar.close()

        results = []
        for seq_id in sorted(outputs):
            token_ids = outputs[seq_id]
            text = self.tokenizer.decode(token_ids, skip_special_tokens=True)
            results.append({"text": text, "token_ids": token_ids})
        return results
```

```python
import argparse
import os
import sys

os.environ["TOKENIZERS_PARALLELISM"] = "false"
sys.path.insert(0, os.path.dirname(os.path.abspath(__file__)))

import torch
from transformers import AutoTokenizer

from llm import LLM
from sampling_params import SamplingParams


def parse_args():
    parser = argparse.ArgumentParser(description="nano-vLLM 单卡示例")
    parser.add_argument("--model_path", type=str, default="models/Qwen3-0.6B")
    parser.add_argument("--max_tokens", type=int, default=128)
    parser.add_argument("--temperature", type=float, default=0.8)
    parser.add_argument("--top_k", type=int, default=20)
    parser.add_argument("--top_p", type=float, default=0.95)
    return parser.parse_args()


def main():
    args = parse_args()
    model_path = os.path.join(os.path.dirname(__file__), args.model_path)

    print("=" * 60)
    print("nano-vLLM 单卡示例")
    print("=" * 60)

    llm = LLM(model_path)
    tokenizer = AutoTokenizer.from_pretrained(model_path, trust_remote_code=True)

    raw_prompts = [
        "你好，请用 200 字介绍一下你自己。",
        "请解释一下什么是 PagedAttention。",
    ]
    prompts = [
        tokenizer.apply_chat_template(
            [{"role": "user", "content": prompt}],
            tokenize=False,
            add_generation_prompt=True,
        )
        for prompt in raw_prompts
    ]

    sampling_params = SamplingParams(
        temperature=args.temperature,
        top_k=args.top_k,
        top_p=args.top_p,
        max_tokens=args.max_tokens,
    )

    outputs = llm.generate(prompts, sampling_params=sampling_params)
    for prompt, output in zip(raw_prompts, outputs):
        print(f"\n[问题] {prompt}")
        print(f"[回答] {output['text']}")


if __name__ == "__main__":
    if torch.cuda.is_available():
        print(f"CUDA: {torch.cuda.get_device_name(0)}")
    main()
```

```python
"""Day 5: 调度、引擎编排与生成循环测试"""

import sys
from types import SimpleNamespace

sys.path.insert(0, '.')

from engine.block_manager import BlockManager
from engine.llm_engine import LLMEngine
from engine.scheduler import Scheduler
from engine.sequence import Sequence, SequenceStatus
from sampling_params import SamplingParams


def make_scheduler(eos: int = 2):
    config = SimpleNamespace(
        max_num_seqs=8,
        max_num_batched_tokens=128,
        eos=eos,
    )
    block_manager = BlockManager(num_blocks=32, block_size=4)
    return Scheduler(config, block_manager)


def test_scheduler_prefill_then_decode():
    scheduler = make_scheduler()

    seq1 = Sequence([10, 11, 12], SamplingParams(max_tokens=4))
    seq2 = Sequence([20, 21], SamplingParams(max_tokens=4))
    seq1.block_size = 4
    seq2.block_size = 4

    scheduler.add_sequence(seq1)
    scheduler.add_sequence(seq2)

    scheduled, is_prefill = scheduler.schedule()
    assert is_prefill is True
    assert len(scheduled) == 2
    assert all(seq.status == SequenceStatus.RUNNING for seq in scheduled)

    finished = scheduler.postprocess(scheduled, [30, 31])
    assert finished == []

    decode_scheduled, is_prefill = scheduler.schedule()
    assert is_prefill is False
    assert len(decode_scheduled) == 2


def test_scheduler_finish_by_eos():
    scheduler = make_scheduler(eos=99)
    seq = Sequence([1, 2, 3], SamplingParams(max_tokens=8))
    seq.block_size = 4
    scheduler.add_sequence(seq)

    scheduled, is_prefill = scheduler.schedule()
    assert is_prefill is True

    finished = scheduler.postprocess(scheduled, [99])
    assert len(finished) == 1
    assert finished[0].status == SequenceStatus.FINISHED
    assert scheduler.is_finished() is True


class DummyTokenizer:
    eos_token_id = 2

    def encode(self, text):
        return [1, 3, 5]

    def decode(self, token_ids, skip_special_tokens=True):
        return " ".join(str(x) for x in token_ids)


class ScriptedRunner:
    def __init__(self):
        self.scripts = {}

    def run(self, sequences, is_prefill):
        outputs = []
        for seq in sequences:
            script = self.scripts.setdefault(seq.seq_id, [42, 2])
            outputs.append(script.pop(0))
        return outputs


def test_llm_engine_generate_loop_with_stub_runner():
    engine = object.__new__(LLMEngine)
    engine.config = SimpleNamespace(eos=2)
    engine.tokenizer = DummyTokenizer()
    engine.block_manager = BlockManager(num_blocks=32, block_size=4)
    scheduler_cfg = SimpleNamespace(max_num_seqs=8, max_num_batched_tokens=128, eos=2)
    engine.scheduler = Scheduler(scheduler_cfg, engine.block_manager)
    engine.model_runner = ScriptedRunner()

    outputs = LLMEngine.generate(
        engine,
        prompts=[[1, 2, 3], [4, 5]],
        sampling_params=SamplingParams(max_tokens=4),
        use_tqdm=False,
    )

    assert len(outputs) == 2
    assert outputs[0]["token_ids"] == [42, 2]
    assert outputs[1]["token_ids"] == [42, 2]
```
